/* =========================================================
 * 1. OpenMV 图像范围配置
 * =========================================================
 * 当前 OpenMV 端使用的是 QVGA(320x240)，并发送目标中心点坐标：
 * X: 0~319
 * Y: 0~239
 *
 * 如果以后改成 QQVGA / VGA / 其他分辨率，
 * 只需要改这里即可。
 */
#define IMAGE_MAX_X                 320.0f
#define IMAGE_MAX_Y                 240.0f

#define IMAGE_CENTER_X              (IMAGE_MAX_X * 0.5f)
#define IMAGE_CENTER_Y              (IMAGE_MAX_Y * 0.5f)

/* =========================================================
 * 2. 舵机通道配置
 * =========================================================
 * 当前使用 TIM2 的两个 PWM 通道：
 * X 轴（水平） -> TIM2_CH1
 * Y 轴（俯仰） -> TIM2_CH2
 *
 * 如果以后换定时器或者换引脚，
 * 优先改这里，并同步检查 CubeMX 对应 PWM 配置。
 */
#define SERVO_X_CHANNEL             TIM_CHANNEL_1
#define SERVO_Y_CHANNEL             TIM_CHANNEL_2

/* =========================================================
 * 3. 舵机角度与脉宽配置
 * =========================================================
 * SERVO_FULL_ANGLE_DEG：
 * 普通 180 度舵机通常保持 180.0f 即可。
 *
 * SERVO_PULSE_MIN_US / MAX_US：
 * 按舵机手册修改脉宽范围。
 * 常见值为 1000~2000us，也有些舵机是 500~2500us。
 */
#define SERVO_FULL_ANGLE_DEG        180.0f
#define SERVO_PULSE_MIN_US          1000.0f
#define SERVO_PULSE_MAX_US          2000.0f

/* =========================================================
 * 4. X / Y 两轴的机械安全角度范围
 * =========================================================
 * 这里的目的不是“让舵机能转满 180 度”，
 * 而是避免云台撞到机械极限。
 *
 * 如果后续发现某个方向快撞到结构件，
 * 直接改对应轴的 MIN / MAX 即可。
 */
#define SERVO_X_MIN_ANGLE           30.0f
#define SERVO_X_MAX_ANGLE           150.0f
#define SERVO_X_START_ANGLE         90.0f

#define SERVO_Y_MIN_ANGLE           10.0f
#define SERVO_Y_MAX_ANGLE           170.0f
#define SERVO_Y_START_ANGLE         90.0f

/* =========================================================
 * 5. 输入滤波参数
 * =========================================================
 * 作用：
 * 对 OpenMV 发来的 X / Y 坐标做低通滤波，
 * 防止识别框轻微抖动时，舵机跟着乱抖。
 *
 * 范围：0~1
 * 越小：越稳，但响应更慢
 * 越大：越灵敏，但更容易抖
 */
#define X_FILTER_FACTOR             0.55f
#define Y_FILTER_FACTOR             0.55f

/* =========================================================
 * 6. 云台平滑输出参数
 * =========================================================
 * SERVO_SMOOTH_FACTOR：
 * 舵机角度平滑系数，范围 0~1
 * 越小：动作更柔和，但会更慢
 * 越大：动作更跟手，但可能更抖
 *
 * SERVO_DEAD_BAND：
 * 角度死区，单位“度”
 * 当前角度距离目标角度足够近时，直接认为到位，
 * 防止舵机在很小范围内不停修正。
 *
 * SERVO_UPDATE_DELAY_MS：
 * 控制循环更新周期。
 */
#define SERVO_SMOOTH_FACTOR         0.40f
#define SERVO_DEAD_BAND             0.20f
#define SERVO_UPDATE_DELAY_MS       5

/* =========================================================
 * 7. 丢目标保持时间（预留）

 * 当前版本搜索逻辑已改为“目标有效/无效”直接切换，

/* =========================================================
 * 8. X 轴（水平）跟随参数

 * X_CENTER_MIN / MAX：
 * 图像中心附近的允许区间。
 * 当目标进入这个范围时，认为“基本居中”，
 * X 轴不再继续修正。
 *
 * X_TRACK_STEP_GAIN_DEG：
 * 每次根据误差修正的角度步进增益。
 * 越大：追得更快

 *
 * X_TRACK_DIRECTION：
 * 如果发现目标在左边，舵机却往反方向转，

 */

#define Y_CENTER_MAX                122.0f
#define Y_TRACK_STEP_GAIN_DEG       12.0f

/* =========================================================
 * 13. 搜索模式参数（主动寻找目标）
 * =========================================================
 * 作用：

 * 扫描方式：

		/* =========================================================

 /* =========================================================

  float filtered_x = IMAGE_CENTER_X;

/**
  * @brief  The application entry point.
  * @retval int
  */
int main(void)
{

  /* USER CODE BEGIN 1 */

  /* USER CODE END 1 */

  /* MCU Configuration--------------------------------------------------------*/

  /* Reset of all peripherals, Initializes the Flash interface and the Systick. */
  HAL_Init();

  /* USER CODE BEGIN Init */

  /* USER CODE END Init */

  /* Configure the system clock */
  SystemClock_Config();

  /* USER CODE BEGIN SysInit */

  /* USER CODE END SysInit */

  /* Initialize all configured peripherals */
  MX_GPIO_Init();
  MX_USART1_UART_Init();
  MX_I2C1_Init();
  MX_USART2_UART_Init();
  MX_TIM2_Init();
  /* USER CODE BEGIN 2 */

/* =========================================================
   * 1. 启动两个 PWM 通道
   * =========================================================
   * X 轴：TIM2_CH1
   * Y 轴：TIM2_CH2
   *
   * 注意：
   * 这里只是启动 PWM 输出，
   * 前提是 CubeMX 中已经把对应通道配置为 PWM 模式。
   */
  HAL_TIM_PWM_Start(&htim2, SERVO_X_CHANNEL);
  HAL_TIM_PWM_Start(&htim2, SERVO_Y_CHANNEL);

  /* 让两个舵机先回到初始角度，防止上电后位置随机 */
  Servo_SetAngle(&htim2, SERVO_X_CHANNEL, current_angle_x);
  Servo_SetAngle(&htim2, SERVO_Y_CHANNEL, SERVO_Y_START_ANGLE);

  /* OLED 初始化 */
  OLED_Init();
  OLED_Refresh();

  /* 开启 USART2 接收中断，每次接收 1 字节 */
  HAL_UART_Receive_IT(&huart2, &rx_data, 1);

  OLED_Clear();
  OLED_ShowString(0, 0, "N:");
  OLED_Refresh();
  /* =========================================================
   * DFPlayer 娴嬭瘯妯″潡鍒濆鍖?
   * =========================================================
   * 1. 缁戝畾 DFPlayer 浣跨敤鐨勪覆鍙?
   * 2. 绛夊緟妯″潡鍜?TF 鍗＄ǔ瀹氫笂鐢?
   * 3. 璁剧疆榛樿闊抽噺锛屼笉鑷姩鎾斁
   */
  DFPlayer_Init(&huart1);
  HAL_Delay(DFPLAYER_STARTUP_DELAY_MS);
  DFPlayer_SetVolume(DFPLAYER_DEFAULT_VOLUME);

  /* =========================================================
   * 2. 目标坐标变量
   * ========================================================= */
  int x = 0, y = 0;                  // 当前解析出的目标坐标
  static int last_x = -1, last_y = -1; // 上一次坐标，用于减少重复处理

 
  /* =========================================================
   * 6. 按键目标选择功能初始化
   * =========================================================
   * 上电默认先选择“厕所”
   * 并把当前模式显示到 OLED，同时发给 OpenMV。
   */
  Update_TargetMode_Display(current_target_mode);
  Send_TargetMode_To_OpenMV(current_target_mode);
	
  /* 切换目标后：云台回中，并进入搜索准备状态 */
  target_angle_x = current_angle_x;
  target_angle_y = SERVO_Y_START_ANGLE;
  search_angle_x = current_angle_x;
  search_direction = 1;
  search_mode = 1;
  target_switch_tick = HAL_GetTick();

  /* USER CODE END 2 */

  /* Infinite loop */
  /* USER CODE BEGIN WHILE */
while (1)
  {
		    /* =========================================================
     * 0. 按键扫描
     * =========================================================
     * 每次循环先检测按钮是否被按下。
     * 如果按下，则切换当前识别目标，并通知 OpenMV。
     */
    Key_Scan_And_Handle();
		
    /* =========================================================
     * 1. 串口接收完成后，解析 OpenMV 发来的坐标
     * =========================================================
     * OpenMV 发送格式：
     * X:123,Y:45
     *
     * 如果没找到目标，则发送：
     * X:-1,Y:-1
     */
    if (uart2_received_flag)
    {
        uart2_received_flag = 0;

        if (sscanf(rx_buffer, "X:%d,Y:%d", &x, &y) == 2)
        {
      /* -----------------------------
      * 1.1 OpenMV 没找到目标
      * ----------------------------- */
      if (x == -1 && y == -1)
      {
          /* =====================================================
          * 丢失目标处理（时间 + 计数双缓冲）
          * ===================================================== */

          /* 如果刚刚还检测到目标（500ms内），认为是短暂丢失 */
          if ((HAL_GetTick() - last_valid_target_tick) < 500)
          {
              /* 什么都不做，保持当前状态 */
          }
          else
          {
              if (search_mode)
              {
                  target_valid = 0;
              }
              else
              {
                  lost_count++;

                  if (lost_count >= LOST_COUNT_MAX)
                  {
                      target_valid = 0;
                      search_mode = 1;
                      target_switch_tick = HAL_GetTick();

                      search_phase = 1;
                      search_center_x = current_angle_x;
                      search_angle_x = current_angle_x;
                      search_direction = (last_x_error > 0) ? 1 : -1;
                  }
              }
          }
      }

      /* -----------------------------
      * 1.2 OpenMV 找到目标
      * ----------------------------- */
      else
      {
          /* 恢复检测 */
          lost_count = 0;

          target_valid = 1;
          last_target_tick = HAL_GetTick();
          last_valid_target_tick = HAL_GetTick();

          search_mode = 0;

          if (x != last_x || y != last_y)
          {
              last_x = x;
              last_y = y;

              /* ===== X轴控制 ===== */
              filtered_x = filtered_x * (1.0f - X_FILTER_FACTOR)
                        + ((float)x * X_FILTER_FACTOR);

              x_error = filtered_x - IMAGE_CENTER_X;
              last_x_error = x_error;

              if ((filtered_x >= X_CENTER_MIN) && (filtered_x <= X_CENTER_MAX))
              {
                  target_angle_x = current_angle_x;
              }
              else
              {
                  target_angle_x += (x_error / IMAGE_CENTER_X)
                                  * X_TRACK_STEP_GAIN_DEG
                                  * X_TRACK_DIRECTION;

                  target_angle_x = Limit_Float(target_angle_x,
                                              SERVO_X_MIN_ANGLE,
                                              SERVO_X_MAX_ANGLE);
              }

              /* ===== Y轴控制 ===== */
              filtered_y = filtered_y * (1.0f - Y_FILTER_FACTOR)
                        + ((float)y * Y_FILTER_FACTOR);

              y_error = filtered_y - IMAGE_CENTER_Y;

              if ((filtered_y >= Y_CENTER_MIN) && (filtered_y <= Y_CENTER_MAX))
              {
                  target_angle_y = current_angle_y;
              }
              else
              {
                  target_angle_y += (y_error / IMAGE_CENTER_Y)
                                  * Y_TRACK_STEP_GAIN_DEG
                                  * Y_TRACK_DIRECTION;

                  target_angle_y = Limit_Float(target_angle_y,
                                              SERVO_Y_MIN_ANGLE,
                                              SERVO_Y_MAX_ANGLE);
              }

              /* OLED */
              OLED_ShowString(0, 0, "X:         ");
              OLED_ShowString(0, 2, "Y:         ");
              OLED_ShowString(0, 4, "TRACKING   ");

              sprintf(tx_buffer, "%d", x);
              OLED_ShowString(12, 0, tx_buffer);

              sprintf(tx_buffer, "%d", y);
              OLED_ShowString(12, 2, tx_buffer);

              OLED_Refresh();
						}
					}
        }
    }

    /* =========================================================
     * 5. 云台更新策略
     * =========================================================
     * 如果当前有目标，或者最近一小段时间内刚见过目标，
     * 就继续保持跟随，避免短时漏检造成动作突变。
     */
    if (target_valid)
    {
        /* -----------------------------
         * 5.1 X 轴平滑更新
         * -----------------------------
         * 这里不是一步跳到 target_angle_x，
         * 而是平滑逼近，防止舵机动作过猛。
         */
        current_angle_x = current_angle_x * (1.0f - SERVO_SMOOTH_FACTOR)
                        + target_angle_x * SERVO_SMOOTH_FACTOR;

        /* 如果已经非常接近目标角度，就直接贴上去 */
        if ((current_angle_x > (target_angle_x - SERVO_DEAD_BAND)) &&
            (current_angle_x < (target_angle_x + SERVO_DEAD_BAND)))
        {
            current_angle_x = target_angle_x;
        }

        /* -----------------------------
         * 5.2 Y 轴平滑更新
         * ----------------------------- */
        current_angle_y = current_angle_y * (1.0f - SERVO_SMOOTH_FACTOR)
                        + target_angle_y * SERVO_SMOOTH_FACTOR;

        if ((current_angle_y > (target_angle_y - SERVO_DEAD_BAND)) &&
            (current_angle_y < (target_angle_y + SERVO_DEAD_BAND)))
        {
            current_angle_y = target_angle_y;
        }

        /* -----------------------------
         * 5.3 输出到两个舵机
         * ----------------------------- */
        Servo_SetAngle(&htim2, SERVO_X_CHANNEL, current_angle_x);
        Servo_SetAngle(&htim2, SERVO_Y_CHANNEL, current_angle_y);
    }
else
{
    /* =====================================================
     * NONE 模式：保持静止
     * ===================================================== */
    if (current_target_mode == TARGET_MODE_NONE)
    {
        target_angle_x = current_angle_x;
        target_angle_y = current_angle_y;

        Servo_SetAngle(&htim2, SERVO_X_CHANNEL, current_angle_x);
        Servo_SetAngle(&htim2, SERVO_Y_CHANNEL, current_angle_y);
    }
    else
    {
        /* =====================================================
         * 搜索模式（主动寻找目标）
         * ===================================================== */

        if (!target_valid && current_target_mode != TARGET_MODE_NONE)
        {
            if (!search_mode)
            {
                search_mode = 1;
                target_switch_tick = HAL_GetTick();
            }
        }

        if (search_mode)
        {
            if ((HAL_GetTick() - target_switch_tick) <= SEARCH_START_DELAY_MS)
            {
                target_angle_x = current_angle_x;
                target_angle_y = current_angle_y;

                Servo_SetAngle(&htim2, SERVO_X_CHANNEL, current_angle_x);
                Servo_SetAngle(&htim2, SERVO_Y_CHANNEL, current_angle_y);
            }
            else
            {
                if ((HAL_GetTick() - search_last_tick) >= SEARCH_UPDATE_MS)
                {
                    search_last_tick = HAL_GetTick();

                    search_angle_x += SEARCH_STEP_DEG * search_direction;

                    if (search_angle_x >= SEARCH_X_RIGHT_ANGLE)
                    {
                        search_angle_x = SEARCH_X_RIGHT_ANGLE;
                        search_direction = -1;
                    }
                    else if (search_angle_x <= SEARCH_X_LEFT_ANGLE)
                    {
                        search_angle_x = SEARCH_X_LEFT_ANGLE;
                        search_direction = 1;
                    }

                    target_angle_x = search_angle_x;
                    target_angle_y = SEARCH_Y_FIXED_ANGLE;

                    current_angle_x = target_angle_x;
                    current_angle_y = target_angle_y;

                    Servo_SetAngle(&htim2, SERVO_X_CHANNEL, current_angle_x);
                    Servo_SetAngle(&htim2, SERVO_Y_CHANNEL, current_angle_y);
                }
            }
        }
    }
}
    HAL_Delay(SERVO_UPDATE_DELAY_MS);

    /* USER CODE END WHILE */

    /* USER CODE BEGIN 3 */
  }
  /* USER CODE END 3 */
}

/*
 * 函数名：Send_TargetMode_To_OpenMV
 * 功能：把当前目标模式通过 USART2 发给 OpenMV
 *
 * 协议格式：
 * TARGET:0\r\n
 * TARGET:1\r\n
 * TARGET:2\r\n
 * TARGET:3\r\n
 * TARGET:4\r\n
 *
 * 说明：
 * 这里使用 USART2，是因为当前工程里 OpenMV 和 STM32 的通信
 * 就是走 huart2 这一条链路。
 */
void Send_TargetMode_To_OpenMV(uint8_t mode)
{
    char cmd[20];

    sprintf(cmd, "TARGET:%d\r\n", mode);
    HAL_UART_Transmit(&huart2, (uint8_t *)cmd, strlen(cmd), HAL_MAX_DELAY);
}

/*
 * 函数名：Key_Scan_And_Handle
 * 功能：扫描五个按钮，并根据按钮结果切换目标模式
 *
 * 当前映射关系：
 *PC0 -> Toilet
 *PC1 -> Exit
 *PE13 -> Office
 *PE14 -> DFPlayer Test
 *PE15 -> None
 *
 * 说明：
 * 1. 当前按钮采用“上拉输入”
 * 2. 所以按下时读到的是 GPIO_PIN_RESET
 * 3. 用统一的消抖时间，防止一次按压触发多次
 */
void Key_Scan_And_Handle(void)
{
    uint32_t now_tick = HAL_GetTick();

    /* 如果还在消抖时间内，就直接返回 */
    if ((now_tick - key_last_tick) < KEY_DEBOUNCE_MS)
    {
        return;
    }

    /* -----------------------------
     * 0. 鎸夐敭锛欴FPlayer Test
     * PE14
     * -----------------------------
     * PE14 褰撳墠鍥哄畾鐢ㄤ綔璇煶娴嬭瘯鎸夐敭锛屼笉鍙備笌鐩爣妯″紡鍒囨崲銆?
     */
    if (HAL_GPIO_ReadPin(KEY_VOICE_PORT, KEY_VOICE_PIN) == GPIO_PIN_RESET)
    {
        key_last_tick = now_tick;
        Handle_Voice_Test_Key();
        return;
    }

    /* -----------------------------
     * 1. 按键：Toilet
     * PC0
     * ----------------------------- */
    if (HAL_GPIO_ReadPin(KEY_PERSON_PORT, KEY_PERSON_PIN) == GPIO_PIN_RESET)
    {
        current_target_mode = TARGET_MODE_TOILET;
        key_last_tick = now_tick;

        Update_TargetMode_Display(current_target_mode);
        Send_TargetMode_To_OpenMV(current_target_mode);

        /* =====================================================
        * 切换目标后：启动搜索流程
        * =====================================================
        * 1. 清除当前目标状态
        * 2. 云台回中
        * 3. 启动搜索模式
        */

        target_valid = 0;                          // 当前没有锁定目标
        search_mode = 1;                           // 进入搜索模式
        target_switch_tick = HAL_GetTick();        // 记录切换时间（用于延时回中）

        search_angle_x = current_angle_x;         // 从原位置开始扫
        search_direction = (last_x_error > 0) ? 1 : -1;                      // 默认向右扫

        search_phase = 1;
        search_center_x = current_angle_x;
        return;
    }
    /* -----------------------------
     * 2. 按键：Exit
     * PC1
     * ----------------------------- */
    if (HAL_GPIO_ReadPin(KEY_BOTTLE_PORT, KEY_BOTTLE_PIN) == GPIO_PIN_RESET)
    {
        current_target_mode = TARGET_MODE_EXIT;
        key_last_tick = now_tick;

        Update_TargetMode_Display(current_target_mode);
        Send_TargetMode_To_OpenMV(current_target_mode);

        target_valid = 0;
        search_mode = 1;
        target_switch_tick = HAL_GetTick();

        search_angle_x = current_angle_x;
        search_direction = (last_x_error > 0) ? 1 : -1;

        search_phase = 1;
        search_center_x = current_angle_x;
        return;
    }
    /* -----------------------------
     * 3. 按键：Office
     * PE13
     * ----------------------------- */
    if (HAL_GPIO_ReadPin(KEY_PEN_PORT, KEY_PEN_PIN) == GPIO_PIN_RESET)
    {
        current_target_mode = TARGET_MODE_OFFICE;
        key_last_tick = now_tick;

        Update_TargetMode_Display(current_target_mode);
        Send_TargetMode_To_OpenMV(current_target_mode);

        target_valid = 0;
        search_mode = 1;
        target_switch_tick = HAL_GetTick();

        search_angle_x = current_angle_x;
        search_direction = (last_x_error > 0) ? 1 : -1;

        search_phase = 1;
        search_center_x = current_angle_x;

        return;
    }
    /* -----------------------------
     * 4. 按键：None
     * PE15
     * ----------------------------- */
      if (HAL_GPIO_ReadPin(KEY_NONE_PORT, KEY_NONE_PIN) == GPIO_PIN_RESET)
      {
          current_target_mode = TARGET_MODE_NONE;
          key_last_tick = now_tick;

          Update_TargetMode_Display(current_target_mode);
          Send_TargetMode_To_OpenMV(current_target_mode);

          /* =====================================================
          * NONE 模式：停止搜索，锁定当前位置
          * ===================================================== */
          target_valid = 0;
          search_mode = 0;
          lost_count = 0;

          /* 把目标角度锁到当前位置，防止继续运动 */
          target_angle_x = current_angle_x;
          target_angle_y = current_angle_y;

          Servo_SetAngle(&htim2, SERVO_X_CHANNEL, current_angle_x);
          Servo_SetAngle(&htim2, SERVO_Y_CHANNEL, current_angle_y);

          return;
      }
    }

/*
 * 鍑芥暟鍚嶏細Handle_Voice_Test_Key
 * 鍔熻兘锛氬鐞?PE14 璇煶娴嬭瘯鎸夐敭鐨勭姸鎬佹満
 *
 * 鐘舵€佽鍒欙細
 * 1. 棣栨鎸変笅 -> 鎾斁 MP3/0001.mp3
 * 2. 绗簩娆℃寜涓?-> 暂停
 * 3. 绗笁娆℃寜涓?-> 继续
 *
 * 璇存槑锛?
 * 褰撳墠鍙仛娴嬭瘯鐗堬紝涓嶄笌鐩爣妯″紡鍜岃窡闅忛€昏緫鑰﹀悎銆?
 * 鍚庣画濡傛灉瑕佹墿灞曠郴缁熻闊虫彁绀猴紝
 * 浼樺厛浠庤繖閲岀殑涓氬姟璋冪敤鍏ュ彛寮€濮嬫暣鐞嗗嵆鍙€?
 */
void Handle_Voice_Test_Key(void)
{
    if (!voice_started)
    {
        DFPlayer_PlayMp3(DFPLAYER_TEST_MP3_INDEX);
        voice_started = 1;
        voice_paused = 0;
        return;
    }

    if (!voice_paused)
    {
        DFPlayer_Pause();
        voice_paused = 1;
    }
    else
    {
        DFPlayer_Resume();
        voice_paused = 0;
    }
}

float Limit_Float(float value, float min, float max)
{
    if (value < min) return min;
    if (value > max) return max;
    return value;
}

void Servo_SetAngle(TIM_HandleTypeDef *htim, uint32_t channel, float angle)
{
    float pulse;

    /* 统一做“逻辑角度 -> PWM 脉宽”映射。
     * 以后如果换舵机，优先修改配置区里的 SERVO_PULSE_MIN_US / SERVO_PULSE_MAX_US。
     */

    // 先限制在 0~180 度
    if (angle < 0.0f) angle = 0.0f;
    if (angle > SERVO_FULL_ANGLE_DEG) angle = SERVO_FULL_ANGLE_DEG;

    // 先用更保守的 1000~2000us 测试
    pulse = SERVO_PULSE_MIN_US +
            (angle / SERVO_FULL_ANGLE_DEG) * (SERVO_PULSE_MAX_US - SERVO_PULSE_MIN_US);

    __HAL_TIM_SET_COMPARE(htim, channel, (uint16_t)pulse);
}

void HAL_UART_RxCpltCallback(UART_HandleTypeDef *huart)
{
    if (huart->Instance == USART2)
    {
        if (rx_data != '\r' && rx_data != '\n')
        {
            if (rx_index < sizeof(rx_buffer) - 1)
            {
                rx_buffer[rx_index++] = rx_data;
            }
        }
        else
        {
            if (rx_index > 0)   // 防止连续收到 \r\n 时空触发
            {
                rx_buffer[rx_index] = '\0';
                rx_index = 0;
                uart2_received_flag = 1;
            }
        }

        HAL_UART_Receive_IT(&huart2, &rx_data, 1);
    }
}
